 - normalize()
 - add( Vector $vec )
 - sub( Vector $vec )
 - opposite()
 - scalarProduct ( $scalar )
 - dotProduct( Vector $vec )
 - cos ( Vector $vec )
 - crossProduct( Vector $vec )
